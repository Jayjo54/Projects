function getResults(questionObj, computerChoice) {
  if (computerChoice === questionObj.answer) {
    return "The computer's choice is correct!";
  } else {
    return `The computer's choice is wrong. The correct answer is: ${questionObj.answer}`;
  }
}

function getRandomQuestion(arr) {
  let randQuestion = Math.floor(Math.random() * arr.length);
  return arr[randQuestion];
}
function getRandomComputerChoice(arr) {
  let compChoice = Math.floor(Math.random() * arr.length);
  return arr[compChoice];
}
// Example usage:
let sampleQuestion = questions[1]; // Science question
let computerPick = "Venus";       // Wrong answer
console.log(getResults(sampleQuestion, computerPick));

computerPick = "Mars";           // Correct answer
console.log(getResults(sampleQuestion, computerPick));
